\subsection{Linear Independence and Rank}

\addDef{Linear Combination}{
    A linear combination of a set of vectors $\{ \bx_1, \bx_2, \cdots, \bx_n \}$ is given by 
    \begin{equation*}
        \sum_{i} \alpha_i \bx_i,
    \end{equation*}
    where $\alpha_1, \cdots,  \alpha_n$ are scalars. 
}


\addDef{Linear Dependence}{
    A set of vectors $\{ \bx_1, \bx_2, \cdots, \bx_n \}$ are linearly dependent if there exists a set of scalars $\{ \alpha_1, \cdots, \alpha_n \}$ such that 
    \begin{equation*}
        \sum_{i} \alpha_i \bx_i = \mathbf{0}.
    \end{equation*}
}
